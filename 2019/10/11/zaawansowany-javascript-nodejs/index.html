GET

app.get('/get', function (req, res) {
  res.send('-> get')
})

curl http://localhost:4000/get

POST

app.post('/post', function (req, res) {
  res.send('-> post')
})

curl -d '{"key":"value1}' -X POST http://localhost:4000/post

PUT

app.put('/put', function (req, res) {
  res.send('-> put')
})

curl -d '{"key":"value1}' -X PUT http://localhost:4000/put

DELETE

app.delete('/delete', function (req, res) {
  res.send('-> delete')
})

curl -X DELTE http://localhost:4000/delete
